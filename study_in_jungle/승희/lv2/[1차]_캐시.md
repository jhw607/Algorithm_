from collections import deque

def solution(cacheSize, cities):
    answer = 0
    
    if cacheSize == 0:
        answer = len(cities) * 5
    else:
        cache = deque()
        for city in cities:

            city = city.lower()
            if city in cache:
                answer += 1
                cache.remove(city)
                cache.append(city)
            else:
                answer += 5
                if len(cache) >= cacheSize:
                    cache.popleft()
                    
                cache.append(city)
                
    return answer
